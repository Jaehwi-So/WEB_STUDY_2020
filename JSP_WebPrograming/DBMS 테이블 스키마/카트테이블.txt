CREATE SEQUENCE seq_cart_idx;

CREATE TABLE CART(
	P_IDX NUMBER(3) PRIMARY KEY,
	C_IDX NUMBER(3) PRIMARY KEY,
	CATEGORY VARCHAR(50) NOT NULL,
	P_NUM VARCHAR(50),
	P_NAME VARCHAR(50),
	P_COMPANY VARCHAR(50),
	P_PRICE NUMBER(10),
	P_SALEPRICE NUMBER(10),
	P_IMAGE_S VARCHAR(50),
	P_IMAGE_L VARCHAR(50),
	P_CONTENT CLOB,
	P_DATE DATE
);
