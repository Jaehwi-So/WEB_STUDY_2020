#어떤 이미지로부터 새로운 이미지를 생성할지를 지정. node.js가 설치된 linux 이미지를 기반으로 이미지 생성
FROM node:latest 


#/ mscl/app 디렉토리 생성
RUN mkdir -p /app

# 디렉토리를 work 디렉토리로 설성

WORKDIR /app

# package.json 파일 생성
COPY package.json ./

# 디펜던시 설치
RUN npm install --silent

COPY . .

CMD ["npm", "start"]

EXPOSE 9090
